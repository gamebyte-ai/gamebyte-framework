    x: 20,
    y: 50,
    width: 200,
    height: 24,
    value: 75,

// Listen for value changes

    if (newValue < 25) {
        healthBar.flash(0xff0000); // Low health warning
    }
});

    style: {
        fontSize: 24,

    onComplete: () => game.timeUp()
});

timer.pause();
timer.resume();
timer.addTime(30); // Add 30 seconds bonus
